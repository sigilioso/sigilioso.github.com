Bash memo I

This post is the first reminder of some bash commands/utilities which are easy to forget if you do not use them frequently.

For loops and command result

# Basic for loop
for V in "one two three"
do
    echo $V
done

# Basic loop in one line
for V in "one two three"; do echo $V; done

# `command` get the command result
# Makes a copy of txt files getting a <name>.copy copy
for F in `find -name "*.txt"`
do
    cp $F $F.copy
done

Bash strings manipulation

Substring

STRING="this_is_an_example"

# ${string:position:length}, ${string:position}
echo ${STRING:0:4} # --> this
echo ${STRING:4} # --> _is_an_example

# ${string#substring} deletes the shortest match from the beginning
echo ${STRING#*_} # --> is_an_example

# ${string#substring} deletes the shortest match from the end
echo ${STRING%_*} # --> this_is_an

# ${string##substring} deletes the longest match from the beginning
echo ${STRING##*_} # --> example

# ${string%%substring} deletes the longest match from the end
echo ${STRING%%_*} # --> this

Replacement

STRING="this_is_an_example"

# $string/pattern/replacement} (Only first match)
${STRING/is/AA} # --> thAA_is_an_example

# $string//pattern/replacement} (All matches)
${STRING//is/AA} # --> thAA_AA_an_example
